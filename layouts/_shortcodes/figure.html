{{/* Extending upon [https://github.com/gohugoio/hugo/blob/master/tpl/tplimpl/embedded/templates/_shortcodes/figure.html] */}} {{/* Help from [https://discourse.gohugo.io/t/has-anyone-a-webp-shortcode/35391] */}} {{- $path := .Get "src" -}} {{- $link := .Get "link" -}} {{- $target := .Get "target" -}} {{- $rel := .Get "rel" -}} {{- $class := .Get "class" -}} {{- $alt := .Get "alt" -}} {{- $attr := .Get "attr" -}} {{- $attrlink := .Get "attrlink" -}} {{- $title := .Get "title" -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- $caption := .Get "caption" -}} {{- $loading := .Get "loading" -}} {{- $msg1 := "The %q shortcode requires a parameter named %q. See %s" -}} {{- $msg2 := "The resource %q passed to %q is not an image. See %s" -}} {{- $msg3 := "The resource %q passed to %q could not be found. See %s" -}} {{/* Determine which rendering path applies */}} {{- $isHTTP := findRE `http[s]{0,1}://` $path 1 -}} {{- $isAbsPath := findRE `^\/.*` $path 1 -}} {{/* These will be populated only in the local-resource branch */}} {{- $jpg := "" -}} {{- $webp := "" -}} {{- $useResource := false -}} {{- if not (or $isHTTP $isAbsPath) -}} {{- if not $path }}{{ errorf $msg1 .Name "path" .Position }}{{ end -}} {{- with $i := resources.Get $path -}} {{- if eq $i.MediaType.MainType "image" -}} {{- if not $width -}}{{- $width = $i.Width -}}{{- end -}} {{- if not $height -}}{{- $height = $i.Height -}}{{- end -}} {{- $fitOptionsToJpg := printf "%vx%v jpg" $width $height -}} {{- $fitOptionsToWebp := printf "%vx%v webp" $width $height -}} {{- if eq $i.MediaType "image/webp" -}} {{- $jpg = $i.Fit $fitOptionsToJpg -}} {{- $webp = $i -}} {{- else if eq $i.MediaType "image/jpeg" -}} {{- $webp = $i.Fit $fitOptionsToWebp -}} {{- $jpg = $i -}} {{- else -}} {{- $jpg = $i.Fit $fitOptionsToJpg -}} {{- $webp = $i.Fit $fitOptionsToWebp -}} {{- end -}} {{- $useResource = true -}} {{- else -}} {{- errorf $msg2 $path $.Name $.Position -}} {{- end -}} {{- else -}} {{- errorf $msg3 $path .Name .Position -}} {{- end -}} {{- end -}} {{/* Resolve the src for non-resource paths */}} {{- $src := $path -}} {{- if $isAbsPath -}} {{- $src = printf "%s%s" .Site.BaseURL (strings.TrimLeft "/" $path) -}} {{- end -}} {{/* Shared alt text logic */}} {{- $altText := "" -}} {{- if $alt -}} {{- $altText = $alt -}} {{- else if $caption -}} {{- $altText = $caption | markdownify | plainify -}} {{- end -}} {{- if $link -}} {{- end -}} {{- if $useResource -}} {{- else -}} {{- end -}} {{- if $link }}{{ end -}} {{- if or $title $caption $attr -}} {{- end -}}
