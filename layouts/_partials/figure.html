{{/* layouts/_shortcodes/figure.html */}} {{/* Extended figure shortcode. Behaves identically to the standard Hugo figure shortcode when used standalone, but gains thumbnail-generation behaviour when rendered inside a {{< gallery >}} shortcode. Standard parameters (unchanged from Hugo built-in): src, link, target, rel, class, alt, attr, attrlink, title, width, height, caption, loading Gallery context (set via $.Page.Store by gallery.html): "galleryThumbnail" — dimension in px (e.g. "300") for thumbnail generation. "galleryId" — used by gallery.js for lightbox grouping. When in gallery context AND src is a local Hugo asset: - A square thumbnail is generated via .Fill (Smart crop). - The full-resolution image permalink is stored in data-fullsrc. - gallery.js reads data-fullsrc for the lightbox; img src stays low-res. For remote (http/https) and static (leading /) paths inside a gallery: - No processing is possible; src is used for both thumbnail and lightbox. References: https://github.com/gohugoio/hugo/blob/master/tpl/tplimpl/embedded/templates/_shortcodes/figure.html https://discourse.gohugo.io/t/has-anyone-a-webp-shortcode/35391 */}} {{- $path := .Get "src" -}} {{- $link := .Get "link" -}} {{- $target := .Get "target" -}} {{- $rel := .Get "rel" -}} {{- $class := .Get "class" -}} {{- $alt := .Get "alt" -}} {{- $attr := .Get "attr" -}} {{- $attrlink := .Get "attrlink" -}} {{- $title := .Get "title" -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- $caption := .Get "caption" -}} {{- $loading := .Get "loading" -}} {{- $msg1 := "The %q shortcode requires a parameter named %q. See %s" -}} {{- $msg2 := "The resource %q passed to %q is not an image. See %s" -}} {{- $msg3 := "The resource %q passed to %q could not be found. See %s" -}} {{/* Determine path type. */}} {{- $isHTTP := findRE `http[s]{0,1}://` $path 1 -}} {{- $isAbsPath := findRE `^\/.*` $path 1 -}} {{/* ----------------------------------------------------------------------- */}} {{/* Gallery context: check if parent shortcode is a gallery. */}} {{/* ----------------------------------------------------------------------- */}} {{- $inGallery := false -}} {{- $galleryMode := "grid" -}} {{- if .Parent -}} {{- if eq .Parent.Name "gallery" -}} {{- $inGallery = true -}} {{- $galleryMode = .Parent.Get "mode" | default ($.Site.Params.gallery.layout | default "grid") -}} {{- end -}} {{- end -}} {{/* Derive thumbnail size from Site config since Page.Store isn't available */}} {{- $thumbnailSize := $.Site.Params.gallery.width | default "300" -}} {{- $isSVG := eq (lower (path.Ext $path)) ".svg" -}} {{/* Variables populated in the local-asset branch. */}} {{- $jpg := "" -}} {{- $webp := "" -}} {{- $useResource := false -}} {{/* Full-resolution permalink — populated for local assets in gallery context. */}} {{- $fullSrc := "" -}} {{/* Thumbnail img src — defaults to $path; overridden for local assets. */}} {{- $thumbSrc := "" -}} {{- if not (or $isHTTP $isAbsPath) -}} {{/* ------------------------------------------------------------------- */}} {{/* Local asset path: process via Hugo image pipeline. */}} {{/* ------------------------------------------------------------------- */}} {{- if not $path }}{{ errorf $msg1 .Name "src" .Position }}{{ end -}} {{- with $i := resources.Get $path -}} {{- if eq $i.MediaType.MainType "image" -}} {{- $thumb := "" -}} {{/* 1. Generate target geometry based on layout context */}} {{- if $inGallery -}} {{- $dim := $thumbnailSize | default "300" -}} {{- if eq $galleryMode "masonry" -}} {{- $thumb = $i.Resize (printf "%sx" $dim) -}} {{- else -}} {{- $thumb = $i.Fill (printf "%sx%s Smart" $dim $dim) -}} {{- end -}} {{- $fullSrc = $i.RelPermalink -}} {{- else -}} {{/* Standalone mode: fit within provided or natural bounds */}} {{- $w := $width | default $i.Width -}} {{- $h := $height | default $i.Height -}} {{- $thumb = $i.Fit (printf "%vx%v" $w $h) -}} {{- end -}} {{/* 2. Update width/height safely from the resized output */}} {{- if not $width -}}{{- $width = $thumb.Width -}}{{- end -}} {{- if not $height -}}{{- $height = $thumb.Height -}}{{- end -}} {{/* 3. Single shared conversion block targeting the unified shape */}} {{- $fitJpg := printf "%vx%v jpg" $thumb.Width $thumb.Height -}} {{- $fitWebp := printf "%vx%v webp" $thumb.Width $thumb.Height -}} {{- if eq $thumb.MediaType "image/webp" -}} {{- $jpg = $thumb.Fit $fitJpg -}} {{- $webp = $thumb -}} {{- else if eq $thumb.MediaType "image/jpeg" -}} {{- $webp = $thumb.Fit $fitWebp -}} {{- $jpg = $thumb -}} {{- else -}} {{- $jpg = $thumb.Fit $fitJpg -}} {{- $webp = $thumb.Fit $fitWebp -}} {{- end -}} {{- $thumbSrc = $webp.RelPermalink -}} {{- $useResource = true -}} {{- else -}} {{- errorf $msg2 $path $.Name $.Position -}} {{- end -}} {{- else -}} {{- errorf $msg3 $path .Name .Position -}} {{- end -}} {{- end -}} {{/* Resolve the src for non-resource paths (static / remote). */}} {{- $src := $path -}} {{- if $isAbsPath -}} {{- $src = printf "%s%s" .Site.BaseURL (strings.TrimLeft "/" $path) -}} {{- end -}} {{- if not $thumbSrc -}}{{- $thumbSrc = $src -}}{{- end -}} {{/* Fallback: use src as thumbnail. */}} {{/* Shared alt text — priority: explicit alt → title → caption → filename. */}} {{- $altText := "" -}} {{- if $alt -}} {{- $altText = $alt -}} {{- else if $title -}} {{- $altText = $title -}} {{- else if $caption -}} {{- $altText = $caption | markdownify | plainify -}} {{- else -}} {{- $altText = path.Base $path | strings.TrimSuffix (path.Ext $path) -}} {{/* Filename without extension as last resort. */}} {{- end -}} {{/* In gallery context, default loading to "lazy" unless the caller overrides it. Gallery images are almost always below the fold. */}} {{- if and $inGallery (not $loading) -}} {{- $loading = "lazy" -}} {{- end -}} {{/* data-fullsrc: present only when inside a gallery AND a full-res URL is available (local assets only). gallery.js reads this for the lightbox URL. Falls back to the `link` param if provided (manual override). */}} {{- $dataFullSrc := "" -}} {{- if $inGallery -}} {{- if $fullSrc -}} {{- $dataFullSrc = $fullSrc -}} {{- else if $link -}} {{- $dataFullSrc = $link -}} {{/* Manual full-res override via link param. */}} {{- end -}} {{- end -}} {{/* Schema.org ImageObject markup for gallery figures. tags are used for machine-readable data that isn't visible to users. contentUrl — canonical full-resolution URL for crawlers. name — image title (maps to figure's title param). description — caption text (maps to figure's caption param). */}} {{- $schemaContentUrl := $dataFullSrc | default $thumbSrc -}}
